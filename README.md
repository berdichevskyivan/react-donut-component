## Install

```
npm install react-donut-component
```

## Usage

> [Try it out in the browser][code-sandbox-try-it-out]

```jsx
import React from 'react'
import { Donut } from 'react-donut-component' 

const App = () => (  
   <Donut>70</Donut>
);

```
